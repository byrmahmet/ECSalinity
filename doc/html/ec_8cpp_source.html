165 void EC_Salinity::_Salinity(float temp)

177  float b1 = -0.0056;

183  float c0 = 0.6766097;

186  float c3 = -0.00000069698;
187  float c4 = 0.0000000010031;
188 
189  r = uS / 42900;
190 
191  r /= (c0 + temp * (c1 + temp * (c2 + temp * (c3 + temp * c4))));
192 
193  r2 = sqrtf(r);
194  ds = b0 + r2 * (b1 + r2 * (b2 + r2 * (b3 + r2 * (b4 + r2 * b5))));

197  salinityPPT = a0 + r2 * (a1 + r2 * (a2 + r2 * (a3 + r2 * (a4 + r2 * a5)))) + ds;
