61 float EC_Salinity::measureEC(float tempCoefficient)
62 {
63  if (!tempC)
64  {
65  measureTemp();
66  }
67 
68  _write_register(EC_TEMPCOEF_REGISTER, tempCoefficient);
69  _send_command(EC_MEASURE_EC);
70  delay((getAccuracy() * EC_EC_MEASURMENT_TIME));
71  mS = _read_register(EC_MS_REGISTER);
72  if (mS == mS)
73  {
74  PPM_500 = mS * 500;
75  PPM_640 = mS * 640;
76  PPM_700 = mS * 700;
77  uS = mS * 1000;
78  S = mS / 1000;
79 
80  salinityPSU = _read_register(EC_SALINITY_PSU);
81  salinityPPT = salinityPSU * 1.004715;
82  salinityPPM = salinityPPT * 1000;
83  }
84 
85  return mS;
86 }

355 void EC_Salinity::_write_register(byte reg, float f)
356 {
357  byte b[5];
358  float f_val = f;
359  b[0] = reg;
360  b[1] = *((uint8_t *)&f_val );
361  b[2] = *((uint8_t *)&f_val + 1);
362  b[3] = *((uint8_t *)&f_val + 2);
363  b[4] = *((uint8_t *)&f_val + 3);
364  Wire.beginTransmission(EC_SALINITY);
365  Wire.write(b, 5);
366  Wire.endTransmission();
367  delay(10);
368 }
