64 float EC_Salinity::measureEC(float tempCoefficient)
65 {
66  RunningMedian conductivityMedian = RunningMedian(accuracy);
67  RunningMedian tempCMedian = RunningMedian(accuracy);
68  int middleThird = accuracy / 3;
69 
70  int i = accuracy;
71  while(i--)
72  {
73  _measureEC();
74  conductivityMedian.add(_conductivity);
75  tempCMedian.add(tempC);
76  }
77 
78  mS = 1000 / (conductivityMedian.getAverage(middleThird) * K);
79 
80  tempF = ((tempCMedian.getAverage(middleThird) * 9) / 5) + 32;
81 
82  if (useTempCompensation)
83  {
84  if (tempCompensationConstant == -1)
85  {
86  mS = mS / (1 + tempCoefficient * (tempCMedian.getAverage(middleThird) - 25.0));
87  }
88  else
89  {
90  mS = mS / (1 + tempCoefficient * (tempCompensationConstant - 25.0));
91  }
92  }
93 
94  if (useDualPointCalibration)
95  {
96  float referenceRange = _referenceHigh - _referenceLow;
97  float readingRange = _readingHigh - _readingLow;
98 
99  mS = (((mS - _readingLow) * referenceRange) / readingRange) + _referenceLow;
100  }
101 
102  PPM_500 = mS * 500;
103  PPM_640 = mS * 640;
104  PPM_700 = mS * 700;
105  uS = mS * 1000;
106  S = mS / 1000;
107 
108  _salinity(tempCMedian.getAverage(middleThird));
109 
110  return mS;
111 }

122 void EC_Salinity::calibrateProbe(float solutionEC, float tempCoef, int numberProbeSamples)
123 {
124  float ec;
125  RunningMedian conductivityMedian = RunningMedian(numberProbeSamples);
126  RunningMedian tempCMedian = RunningMedian(numberProbeSamples);
127  int middleThird = numberProbeSamples / 3;
128 
129  int i = numberProbeSamples;
130  while(i--)
131  {
132  _measureEC();
133  conductivityMedian.add(_conductivity);
134  tempCMedian.add(tempC);
135  delay(250);

137 
138  ec = solutionEC * (1 + tempCoef * (tempCMedian.getAverage(middleThird) - 25.0));

185 void EC_Salinity::_salinity(float temp)
186 {
187  float r, ds, r2;
188 
189  float a0 = 0.008;
190  float a1 = -0.1692;
191  float a2 = 25.3851;
192  float a3 = 14.0941;
193  float a4 = -7.0261;
194  float a5 = 2.7081;
195 
196  float b0 = 0.0005;
197  float b1 = -0.0056;
198  float b2 = -0.0066;
199  float b3 = -0.0375;
200  float b4 = 0.0636;
201  float b5 = -0.0144;
202 
203  float c0 = 0.6766097;
204  float c1 = 0.0200564;
205  float c2 = 0.0001104259;
206  float c3 = -0.00000069698;
207  float c4 = 0.0000000010031;
208 
209  r = uS / 42900;
210 
211  r /= (c0 + temp * (c1 + temp * (c2 + temp * (c3 + temp * c4))));
212 
213  r2 = sqrtf(r);
214  ds = b0 + r2 * (b1 + r2 * (b2 + r2 * (b3 + r2 * (b4 + r2 * b5))));
215  ds = ds * ((temp - 15.0) / (1.0 + 0.0162 * (temp - 15.0)));
216 
217  // the formula to convert between EC and PSU is not valid for values
218  // below 2 and above 42, or for temperatures below -2C or above 35C.
219  salinityPPT = -1;
220  salinityPPM = -1;
221  salinityPSU = a0 + r2 * (a1 + r2 * (a2 + r2 * (a3 + r2 * (a4 + r2 * a5)))) + ds;
222 
223  if (salinityPSU < 2 || salinityPSU > 42)
224  {
225  salinityPSU = -1;
226  return;
227  }
228 
229  if (tempC < -2 || tempC > 35)
230  {
231  salinityPSU = -1;
232  return;
233  }
234 
235  salinityPPT = salinityPSU * 1.004715;
236  salinityPPM = salinityPPT * 1000;
237 }
